Átlag számítás:
SELECT
    AVG(`jegy`)
FROM
    `ellenorzo`
WHERE
    1

Alias:
SELECT
    AVG(`jegy`) AS `jegyek atlaga`
FROM
    `ellenorzo`
WHERE
    1;

Csoportosítás:
SELECT
    `tantargy_neve` AS `A tantárgy neve`,
    AVG(`jegy`) AS `jegyek atlaga`
FROM
    `ellenorzo`
WHERE
    1
GROUP BY
    `tantargy_neve`

Lekérdezések:

Betű szerinti lekérdezés:
SELECT
    *
FROM
    ellenorzo
WHERE
    tantargy_neve LIKE "m%";

SELECT
    *
FROM
    ellenorzo
WHERE
    tantargy_neve LIKE "%e%";

Nemzetiség szerinti lekérdezés és betűrendbe rakás:
SELECT
    `nev`
FROM
    `versenyzo`
WHERE
    nemzetiseg = 'HUN'
ORDER BY 
	`nev`;
